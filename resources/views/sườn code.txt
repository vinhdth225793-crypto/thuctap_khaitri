CREATE TABLE khoa_hoc_giao_vien (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    khoa_hoc_id BIGINT UNSIGNED NOT NULL,
    giao_vien_id BIGINT UNSIGNED NOT NULL,
    vai_tro ENUM('giao_vien_chinh','giao_vien_module','tro_giang') NOT NULL DEFAULT 'giao_vien_module',
    ghi_chu TEXT NULL,
    created_by BIGINT UNSIGNED NOT NULL,
    created_at TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_khoa_hoc_giao_vien_khoa FOREIGN KEY (khoa_hoc_id) REFERENCES khoa_hoc(id),
    CONSTRAINT fk_khoa_hoc_giao_vien_gv FOREIGN KEY (giao_vien_id) REFERENCES giao_vien(id),
    CONSTRAINT uq_khoa_hoc_giao_vien UNIQUE (khoa_hoc_id, giao_vien_id)
);

CREATE TABLE phan_cong_module_giang_vien (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    khoa_hoc_id BIGINT UNSIGNED NOT NULL,
    module_hoc_id BIGINT UNSIGNED NOT NULL,
    giao_vien_id BIGINT UNSIGNED NOT NULL,
    ngay_phan_cong DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    trang_thai ENUM('cho_xac_nhan','da_nhan','tu_choi') NOT NULL DEFAULT 'da_nhan',
    ghi_chu TEXT NULL,
    created_by BIGINT UNSIGNED NOT NULL,
    created_at TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_phan_cong_module_khoa FOREIGN KEY (khoa_hoc_id) REFERENCES khoa_hoc(id),
    CONSTRAINT fk_phan_cong_module_module FOREIGN KEY (module_hoc_id) REFERENCES module_hoc(id),
    CONSTRAINT fk_phan_cong_module_gv FOREIGN KEY (giao_vien_id) REFERENCES giao_vien(id),
    CONSTRAINT uq_module_giang_vien UNIQUE (module_hoc_id, giao_vien_id)
);
